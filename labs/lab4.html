function checkConfirm() {
    const result = confirm("Нажмите OK или Отмена");
    return result ? "Вы нажали OK" : "Вы нажали Отмена";
}

2 Редактор под себя

function greetUser() {
    const surname = prompt("Введите вашу фамилию:", " ");
    if (surname) {
        alert(`Здравствуйте, ${surname}!`);
    }
}

3 Скажи мне кто ты и я скажу тебе кто ты

function userInterview() {
    const name = prompt("Ваше имя?", "Алексей");
    const age = prompt("Сколько лет?", "20");
    const mood = prompt("Как настроение?", "отличное");
    
    document.write(`<p>Здравствуй, ${name}</p>`);
    document.write(`<p>Тебе ${age} лет</p>`);
    document.write(`<p>И у тебя ${mood} настроение!</p>`);
}

4 ИЗ строки в число

function convertAndSum() {
    const x = prompt("Введите целое число x:", "10");
    const y = prompt("Введите дробное число y:", "5.5");
    
    const integerX = parseInt(x, 10);
    const floatY = parseFloat(y);
    const result = integerX + floatY;
    
    return `x = ${x}, y = ${y}\n` +
           `integerX = ${integerX}, floatY = ${floatY}\n` +
           `Результат сложения: ${result}`;
}
