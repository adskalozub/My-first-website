function isPrime(num) { if (num <= 1) return false; if (num <= 3) return true; if (num % 2 === 0 || num % 3 === 0) return false; for (let i = 5; i * i <= num; i += 6) { if (num % i === 0 || num % (i + 2) === 0) return false; } return true; } function findNextPrime(num) { if (!isPrime(num)) return 0; let next = num + 1; while (!isPrime(next)) { next++; } return next; }

Код 2:

function gcd(a, b) { while (b !== 0) { let temp = b; b = a % b; a = temp; } return a; } function lcm(a, b) { return (a * b) / gcd(a, b); } function lcmOfRange(n) { let result = 1; for (let i = 2; i <= n; i++) { result = lcm(result, i); } return result; }
